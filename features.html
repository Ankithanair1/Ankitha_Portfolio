#include int main() { float a, b; char op; printf("Enter expression (a + b): "); scanf("%f %c %f", &a, &op, &b); switch(op) { case '+': printf("Result = %.2f", a + b); break; case '-': printf("Result = %.2f", a - b); break; case '*': printf("Result = %.2f", a * b); break; case '/': printf("Result = %.2f", a / b); break; default: printf("Invalid operator"); } return 0; }
